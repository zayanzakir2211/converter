ðŸ”¹ Binary Conversion

JS Text to Binary 
// Text to Binary
function textToBinary(text) {
  return text.split('').map(char => 
    char.charCodeAt(0).toString(2).padStart(8, '0')
  ).join(' ');
}

console.log(textToBinary("Hi")); // 01001000 01101001

// Binary to Text
function binaryToText(binary) {
  return binary.split(' ').map(bin => 
    String.fromCharCode(parseInt(bin, 2))
  ).join('');
}

console.log(binaryToText("01001000 01101001")); // Hi

ðŸ”¹ Hex Conversion

JS Text to Hex 
// Text to Hex
function textToHex(text) {
  return Array.from(text).map(c => 
    c.charCodeAt(0).toString(16).padStart(2, '0')
  ).join(' ');
}

console.log(textToHex("Hello")); // 48 65 6c 6c 6f

// Hex to Text
function hexToText(hex) {
  return hex.split(' ').map(h => 
    String.fromCharCode(parseInt(h, 16))
  ).join('');
}

console.log(hexToText("48 65 6c 6c 6f")); // Hello

ðŸ”¹ Hash Generation (SHA-256)

JS Generate SHA-256 Hash 
// Using Web Crypto API
async function sha256(message) {
  const msgBuffer = new TextEncoder().encode(message);
  const hashBuffer = await crypto.subtle.digest('SHA-256', msgBuffer);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
  return hashHex;
}

// Usage
sha256("Hello World").then(hash => {
  console.log(hash); // a591a6d40bf420404a011733cfb7b190...
});
